```python
from collections import deque

def solution(queue1, queue2):
    answer = 0
    q1 = deque(queue1)
    q2 = deque(queue2)
    
    sum1 = sum(q1)
    sum2 = sum(q2)
    sum_total = sum1 + sum2
    
    if sum_total % 2 != 0:
        return -1
    
    while answer <= 2 * (len(q1) + len(q2)):
        if sum1 == sum2:
            return answer
        elif sum1 > sum2:
            p = q1.popleft()
            sum1 -= p
            sum2 += p
            q2.append(p)
        elif sum1 < sum2:
            p = q2.popleft()
            sum1 += p 
            sum2 -= p
            q1.append(p)
        answer += 1
    return -1
```
